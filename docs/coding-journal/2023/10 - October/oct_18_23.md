###### Graph BFS traversal
```python
def BFS(graph):
    queue = []
    visited = []
    traversal = []
    for vertex in graph:
        if vertex not in visited:
            queue.append(vertex)
            while queue:
                node = queue.pop(0)
                if node not in visited:
                    visited.append(node)
                    traversal.append(node)

                    for neighbor in graph[node]:
                        if neighbor not in visited:
                            queue.append(neighbor)
    return traversal


def main():
    graph = {
        'A': ['B', 'C'],
        'B': ['D'],
        'C': [],
        'D': ['E'],
        'E': []
    }
    print(BFS(graph))


if __name__ == "__main__":
    main()

```

###### Graph DFS traversal

```python
def DFS(graph):
    stack = []
    visited = []
    traversal = []
    for vertex in graph:
        if vertex not in visited:
            stack.append(vertex)
            while stack:
                node = stack.pop()
                if node not in visited:
                    visited.append(node)
                    traversal.append(node)
                    for neighbor in graph[node]:
                        if neighbor not in visited:
                            stack.append(neighbor)
    return traversal


def main():
    graph = {
        'A': ['B', 'C'],
        'B': ['D'],
        'C': [],
        'D': ['E'],
        'E': []
    }
    print(DFS(graph))


if __name__ == "__main__":
    main()
```
