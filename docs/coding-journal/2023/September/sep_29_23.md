### BFS traversal

```python
from collections import defaultdict, deque
class Graph:
    def __init__(self):
        self.graph = defaultdict(list)

    def addEdge(self, u, v):
        self.graph[u].append(v)

    # My solution
    def breadth_first_search(self, s): # s is starting vertex
        visited = set()
        queue = [s]
        while queue:
            node = queue.pop(0)
            if node not in visited:
                visited.add(node)
                print(node, end=" ")

            for neighbor in self.graph[node]:
                if neighbor not in visited:
                    queue.append(neighbor)
        print()

    # Geeksforgeeks solution
    def BFS(self, s):
        visited = [False] * (max(self.graph) + 1)
        queue = []
        queue.append(s)
        visited[s] = True
        while queue:
            s = queue.pop(0)
            print(s, end=" ")
            for i in self.graph[s]:
                if visited[i] == False:
                    queue.append(i)
                    visited[i] = True
        print()

def main():
    my_graph = Graph()
```

```python
def dfs_recursive(graph, start):
    visited = set()
    traversal = []
    return dfs_recursive_helper(visited, graph, start, traversal)


def dfs_recursive_helper(visited, graph, start, traversal):
    if start not in visited:
        visited.add(start)
        traversal.append(start)
        for neighbor in graph[start]:
            dfs_recursive_helper(visited, graph, neighbor, traversal)
    return traversal


def dfs(graph, node):
    stack = [node]
    visited = set()
    traversal = []
    while stack:
        current = stack.pop()
        if current not in visited:
            visited.add(current)
            traversal.append(current)
            for neighbor in reversed(graph[current]):
                stack.append(neighbor)
    return traversal
```
