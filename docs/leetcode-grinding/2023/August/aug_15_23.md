### Sort Colors

```
def sort_colors(colors):
    red = white = 0
    blue = len(colors) - 1
    while white <= blue:
        if colors[white] == 0:
            if colors[red] != 0:
                colors[white], colors[red] = colors[red], colors[white]
            white += 1
            red += 1
        elif colors[white] == 1:
            white += 1
        else:
            if colors[blue] != 2:
                colors[blue], colors[white] = colors[white], colors[blue]
            blue -= 1
    return colors


def main():
    colors1 = [2, 1, 1, 2, 0, 1, 2, 0]
    print(sort_colors(colors1))
    colors2 = [0, 2, 1, 2, 2, 1, 2, 0]
    print(sort_colors(colors2))


if __name__ == "__main__":
    main()

### 3Sum

```
def three_sum(nums):
    nums.sort()
    result = []
    n = len(nums)

    # Check for short arrays
    if n < 3:
        return []

    for i in range(n - 2):
        # Skip duplicate numbers
        if i > 0 and nums[i] == nums[i-1]:
            continue

        # Two pointers approach for the rest of the numbers
        start, end = i + 1, n - 1
        while start < end:
            total = nums[i] + nums[start] + nums[end]
            if total == 0:
                result.append([nums[i], nums[start], nums[end]])
                start += 1
                end -= 1
                # Skip duplicates
                while start < end and nums[start] == nums[start - 1]:
                    start += 1
                while start < end and nums[end] == nums[end + 1]:
                    end -= 1
            elif total < 0:
                start += 1
            else:
                end -= 1

    return result


def main():
    print(three_sum([-1, 0, 1, 2, -1, -4]))


if __name__ == "__main__":
    main()
