from typing import Optional
from listnode import ListNode


def merge_two_sorted_lists(head1: Optional[ListNode], head2: Optional[ListNode]) -> Optional[ListNode]:
    node = ListNode(0)
    res = node
    while head1 and head2:
        if head1.val < head2.val:
            node.next = head1
            head1 = head1.next
        else:
            node.next = head2
            head2 = head2.next
        node = node.next

    while head1:
        node.next = head1
        head1 = head1.next
        node = node.next

    while head2:
        node.next = head2
        head2 = head2.next
        node = node.next

    return res.next

  • Linked List Cycle
linked_list_cycle.py
from typing import Optional
from listnode import ListNode


def has_cycle(head: Optional[ListNode]) -> bool:
    fast = head
    slow = head
    while fast and fast.next:
        fast = fast.next.next
        slow = slow.next
        if fast == slow:
            return True
    return False


def main():
    head = ListNode(1)
    head.next = ListNode(2)
    head.next.next = ListNode(3)
    # head.next.next.next = head.next
    head.next.next.next = ListNode(4)
    print(has_cycle(head=head))


if __name__ == "__main__":
    main()

import re


def is_palindrome(s: str) -> bool:
    s = re.sub(r'[\s|\W_]', '', s.strip().lower())
    start = 0
    end = len(s) - 1
    while start < end:
        if s[start] != s[end]:
            return False
        start += 1
        end -= 1
    return True


def is_palindrome_i(s: str) -> bool:
    start = 0
    end = len(s) - 1
    while start < end:
        if not s[start].isalnum():
            start += 1
        elif not s[end].isalnum():
            end -= 1
        elif s[start].lower() == s[end].lower():
            start += 1
            end -= 1
        else:
            return False
    return True

  • Two Sum II
two_sum_ii.py
from typing import List


def two_sum_ii(numbers: List[int], target: int) -> List[int]:
    start = 0
    end = len(numbers) - 1
    while start < end:
        cur_val = numbers[start] + numbers[end]
        if cur_val == target:
            return [start + 1, end + 1]
        elif cur_val < target:
            start += 1
        else:
            end -= 1


def main():
    print(two_sum_ii([2, 7, 11, 15], 9))
    print(two_sum_ii([2, 3, 4], 6))
    print(two_sum_ii([-1, 0], -1))


if __name__ == "__main__":
    main()

three_sum.py
from typing import List


def three_sum(nums: List[int]) -> List[List[int]]:
    nums.sort()
    res = []
    for i in range(len(nums) - 2):
        start = i + 1
        end = len(nums) - 1
        while start < end:
            triplet = [nums[i], nums[start], nums[end]]
            if sum(triplet) == 0 and triplet not in res:
                res.append(triplet)
                start += 1
                end -= 1
            elif sum(triplet) < 0:
                start += 1
            else:
                end -= 1
    return res


def three_sum_i(nums: List[int]) -> List[List[int]]:
    nums.sort()
    res = []
    for i in range(len(nums) - 2):
        if i > 0 and nums[i] == nums[i - 1]:
            continue
        start = i + 1
        end = len(nums) - 1
        while start < end:
            triplet = [nums[i], nums[start], nums[end]]
            if sum(triplet) == 0 and triplet not in res:
                res.append(triplet)
                start += 1
                end -= 1
                while start < end and nums[start] == nums[start - 1]:
                    start += 1
                while start < end and nums[end] == nums[end + 1]:
                    end -= 1
            elif sum(triplet) < 0:
                start += 1
            else:
                end -= 1
    return res


def main():
    print(three_sum([-1, 0, 1, 2, -1, -4]))
    print(three_sum([0, 1, 1]))
    print(three_sum([0, 0, 0]))

    print(three_sum_i([-1, 0, 1, 2, -1, -4]))
    print(three_sum_i([0, 1, 1]))
    print(three_sum_i([0, 0, 0]))


if __name__ == "__main__":
    main()
  • Is Subsequence
is_subsequence.py
def is_subsequence(s: str, t: str) -> bool:
    if not s:
        return True
    if not t:
        return False
    s_ptr = 0
    t_ptr = 0
    while t_ptr < len(t):
        if t[t_ptr] == s[s_ptr]:
            s_ptr += 1
            if s_ptr == len(s):
                return True
        t_ptr += 1
    return s_ptr == len(s)


def main():
    s = "abc"
    t = "ahbgdc"
    print(is_subsequence(s, t))
    print(is_subsequence("acb", t))


if __name__ == "__main__":
    main()
  • Container With Most Water
container_with_most_water.py
from typing import List


def max_area(height: List[int]) -> int:
    res = 0
    start = 0
    end = len(height) - 1
    while start < end:
        cur_width = end - start
        cur_height = min(height[start], height[end])
        cur_area = cur_width * cur_height
        res = max(res, cur_area)
        if height[start] < height[end]:
            start += 1
        else:
            end -= 1
    return res


def main():
    print(max_area([1, 8, 6, 2, 5, 4, 8, 3, 7]))
    print(max_area([1, 1]))


if __name__ == "__main__":
    main()

def length_of_longest_substring(s: str) -> int:
    start = 0
    max_len = 0
    char_dict = {}
    for end in range(len(s)):
        if s[end] in char_dict:
            start = max(start, char_dict[s[end]] + 1)
        char_dict[s[end]] = end
        max_len = max(max_len, end - start + 1)
    return max_len


def length_of_longest_substring_i(s: str) -> int:
    chars = []
    max_len = 0
    for end in range(len(s)):
        if s[end] not in chars:
            chars.append(s[end])
        else:
            while s[end] in chars:
                max_len = max(max_len, len(chars))
                chars = chars[1:]
            chars.append(s[end])
    return max(max_len, len(chars))


def main():
    s1 = "abcabcbb"
    s2 = "bbbbb"
    s3 = "pwwkew"
    s4 = " "
    print(length_of_longest_substring(s1))
    print(length_of_longest_substring_i(s1))

    print(length_of_longest_substring(s2))
    print(length_of_longest_substring_i(s2))

    print(length_of_longest_substring(s3))
    print(length_of_longest_substring_i(s3))

    print(length_of_longest_substring(s4))
    print(length_of_longest_substring_i(s4))

  • Minimum Size Subarray Sum
from typing import List


def min_sub_array_len(target: int, nums: List[int]) -> int:
    start = 0
    res = float("inf")
    cur_sum = 0
    for end in range(len(nums)):
        cur_sum += nums[end]
        while cur_sum >= target:
            res = min(end - start + 1, res)
            cur_sum -= nums[start]
            start += 1
    if res == float("inf"):
        return 0
    return res


def main():
    print(min_sub_array_len(target=7, nums=[2, 3, 1, 2, 4, 3]))
    print(min_sub_array_len(target=4, nums=[1, 4, 4]))
    print(min_sub_array_len(target=11, nums=[1, 1, 1, 1, 1, 1, 1]))
    print(min_sub_array_len(target=11, nums=[1, 2, 3, 4, 5]))


if __name__ == "__main__":

  • Minimum Window Substring
from collections import Counter


def minimum_window(s: str, t: str) -> str:
    if len(t) > len(s):
        return ""

    need = Counter(t)
    match = 0
    l = 0
    start = 0
    window_len = len(s) + 1
    for r, ch in enumerate(s):
        if ch in need:
            need[ch] -= 1
            match += need[ch] == 0
        while match == len(need):
            if window_len > r - l + 1:
                start = l
                window_len = r - l + 1
            removeCh = s[l]
            l += 1
            if removeCh in need:
                match -= need[removeCh] == 0
                need[removeCh] += 1
    return s[start:start + window_len] if window_len <= len(s) else ''


def main():
    print(minimum_window(s="ADOBECODEBANC", t="ABC"))


if __name__ == "__main__":
    main()

  • Repeated DNA Sequences
from typing import List


def find_repeated_dna_sequences(s: str) -> List[str]:
    if len(s) < 10:
        return []

    sub_string = s[:10]
    start = 0
    res = []
    for end in range(10, len(s)):
        if sub_string in s[start + 1:] and sub_string not in res:
            res.append(sub_string)
        start += 1
        sub_string = s[start: start + 10]
    return res


def find_repeated_dna_sequences_i(s: str) -> List[int]:
    if len(s) < 10:
        return []

    seen = set()
    repeated = set()
    for i in range(len(s) - 9):
        sequence = s[i: i + 10]
        if sequence in seen:
            repeated.add(sequence)
        else:
            seen.add(sequence)
    return list(repeated)


def main():
    print(find_repeated_dna_sequences("AAAAACCCCCAAAAACCCCCCAAAAAGGGTTT"))
    print(find_repeated_dna_sequences("AAAAAAAAAAAAA"))

    print(find_repeated_dna_sequences_i("AAAAACCCCCAAAAACCCCCCAAAAAGGGTTT"))
    print(find_repeated_dna_sequences_i("AAAAAAAAAAAAA"))


if __name__ == "__main__":
    main()
