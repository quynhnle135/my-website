Two Sum​

function twoSum(arr, target) {
    let dict = {};
    for (let i = 0; i < arr.length; i++) {
        if ((target - arr[i]) in dict) {
            return [dict[target - arr[i]], i];
        } else {
            dict[arr[i]] = i;
        }
    }
    return [-1, -1];
}

function twoSumBruteForce(arr, target) {
    for (let i = 0; i < arr.length; i++) {
        let curSum = arr[i];
        for (let j = i + 1; j < arr.length; j++) {
            curSum += arr[j];
            if (curSum === target) {
                return [i, j];
            }
            curSum -= arr[j];
        }
    }
    return [-1, -1];
}

Merge Two Sorted Arrays​

function mergeTwoSortedArrays(arr1, arr2) {
    let first = 0;
    let second = 0;
    let arr = [];
    while (first < arr1.length && second < arr2.length) {
        if (arr1[first] < arr2[second]) {
            arr.push(arr1[first]);
            first++;
        } else {
            arr.push(arr2[second]);
            second++;
        }
    }
    while (first < arr1.length) {
        arr.push(arr1[first]);
        first++;
    }
    while (second < arr2.length) {
        arr.push(arr2[second]);
        second++;
    }
    return arr;
} 

console.log(mergeTwoSortedArrays([2, 4, 6], [1, 3, 5, 7]));

function rotateArray(nums, k) {

    reverseArray(0, nums.length - 1, nums);
    console.log(nums);

} 

function reverseArray(start, end, arr) {
    while (start < end) {
        let temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;
        start++;
        end--;
    }
    return arr;
}

function reverseLinkedList(head) {
    let prev = null;
    let cur = head;
    while (cur) {
        let nextNode = cur.next;
        cur.next = prev;
        prev = cur;
        cur = nextNode;
    }
    return prev;
}

function breadthFirstSearch(graph) {
    let queue = [];
    let traversal = [];
    let visited = new Set();
    let firstNode = Object.keys(graph)[0]
    queue.push(firstNode);
    while (queue.length > 0) {
        let node = queue.shift();
        visited.add(node);
        traversal.push(node);
        for (let neighbor of graph[node]) {
            if (!visited.has(neighbor)) {
                queue.push(neighbor);
            }
        }
    }
    return traversal;
}

Number of islands​

function numberOfIslands(grid) {
    let islands = 0;
    for (let row = 0; row < grid.length; row++) {
        for (let col = 0; col < grid[0].length; col++) {
            if (grid[row][col] == "1") {
                islands++;
                dfs(row, col, grid);
            }
        }
    }
    return islands;
}

function dfs(row, col, grid) {
    if (row < 0 || col < 0 || row >= grid.length || col >= grid[0].length || grid[row][col] == "0") {
        return;
    }
    grid[row][col] = "0";
    dfs(row + 1, col, grid);
    dfs(row - 1, col, grid);
    dfs(row, col + 1, grid);
    dfs(row, col - 1, grid);
}

Number of provinces​

function numberOfProvinces(graph) {
    // Convert graph to adjacency list
    var adj_list = {};
    for (let row = 0; row < graph.length; row++) {
        adj_list[parseInt(row)] = [];
        for (let col = 0; col < graph[0].length; col++) {
            if (graph[row][col] === 1) {
                adj_list[row].push(col);
            }
        }
    }
    let provinces = 0;
    let visited = new Set();
    for (let key in adj_list) {
        if (!visited.has(parseInt(key))) {
            depthFirstSearch(key, adj_list, visited);
            provinces++;
            
        }
    }
    console.log(visited);
    return provinces;
}

function depthFirstSearch(node, adj_list, visited) {
    visited.add(node);
    for (let neighbor of adj_list[node]) {
        if (!visited.has(neighbor)) {
            visited.add(neighbor);
            depthFirstSearch(neighbor, adj_list, visited);
        }
    }
}

Maximum depth of a tree (reviewed binary tree)​

class TreeNode {
    constructor(val) {
        this.val = val;
        this.left = null;
        this.right = null;
    }
}

function findMaxDepth(root) {
    if (!root) {
        return 0;
    }
    return 1 + Math.max(findMaxDepth(root.left), findMaxDepth(root.right));
}

Tree Traversal​

class TreeNode {
    constructor(val) {
        this.val = val;
        this.left = null;
        this.right = null;
    }
}

function preOrderTraversal(root) {
    function helper(node, traversal) {
        if (node) {
            traversal.push(node.val);
            helper(node.left, traversal);
            helper(node.right, traversal);
        }
        return traversal;
    } 
    return helper(root, []);
}

function inOrderTraversal(root) {
    function helper(node, traversal) {
        if (node) {
            helper(node.left, traversal);
            traversal.push(node.val);
            helper(node.right, traversal);
        }
        return traversal;
    } 
    return helper(root, []);
}

function postOrderTraversal(root) {
    function helper(node, traversal) {
        if (node) {
            helper(node.left, traversal);
            helper(node.right, traversal);
            traversal.push(node.val);
        }
        return traversal;
    } 
    return helper(root, []);
}

import { BinarySearchTree, Node } from "./BinarySearchTree.js";

function findHeight(root) {
    if (!root) return 0;
    if (!root.left && !root.right) return 0;
    let left = findHeight(root.left);
    let right = findHeight(root.right);
    return 1 + Math.max(left, right);
}

function findKthLargestValueBF(root, k) {
    let traversal = inOrderTraversal(root);
    return traversal[traversal.length - k];
}

function inOrderTraversal(root) {
    function helper(node, traversal) {
        if (node) {
            helper(node.left, traversal);
            traversal.push(node.val);
            helper(node.right, traversal);
        }
        return traversal;
    }
    return helper(root, []);
}

function findKthLargestValue(root, k) {
    let stack = [];
    let cur = root;
    let count = 0;
    while (stack.length > 0 || cur) {
        while (cur) {
            stack.push(cur);
            cur = cur.right;
        }
        cur = stack.pop();
        count++;
        if (count === k) return cur.val;
        cur = cur.left;
    }
    return null;
}

function findAncestor(root, target) {

function helper(node, target, result) {
    if (!node) return false;
    if (node.val === target) return true;
    if (helper(node.left, target, result) || helper(node.right, target.result)) {
        result.push(node.val);
        return true;

    return false;
}
