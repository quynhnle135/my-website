  • Search in BST
class Node {
    constructor(val) {
        this.val = val;
        this.left = null;
        this.right = null;
    }
}

class BinarySearchTree {
    constructor(newValue) {
        this.root = new Node(newValue);
    }

    insert(newValue) {
        if (this.root === null) {
            this.root = new Node(newValue);
            return;
        }

        // start insert from the root
        let curNode = this.root;
        let parentNode;
        while (curNode) {
            parentNode = curNode;
            if (newValue < curNode.val) {
                curNode = curNode.left;
            } else {
                curNode = curNode.right;
            }
        } 

        if (newValue < parentNode.val) {
            parentNode.left = new Node(newValue);
        } else {
            parentNode.right = new Node(newValue);
        }
    }
}

function searchInBST(root, target) {
    if (root === null) {
        return false;
    }
    // start searching from the root
    let curNode = root;
    while (curNode) {
        if (target === curNode.val) {
            return true;
        } else if (target < curNode.val) {
            curNode = curNode.left;
        } else {
            curNode = curNode.right;
        }
    }
    return false;
}

function inOrderTraversal(root) {
    function helper(node, traversal) {
        if (node) {
            helper(node.left, traversal);
            traversal.push(node.val);
            helper(node.right, traversal);
        }
        return traversal;
    }
    return helper(root, []);
}

// find minimum value recursively
function findMinimumValue(root) {
    function helper(node) {
        if (node.left === null) {
            return node.val;
        }
        return helper(node.left);
    }
    return helper(root);
}

// find minimum value iteratively
function findMin(root) {
    if (root === null) {
        return;
    }

    let cur = root;
    while (cur.left) {
        cur = cur.left;
    }
    return cur.val;
}

import { Node, BinarySearchTree } from "./BinarySearchTree.js";


function findKthBiggestElement(root, k) {
    let traversal = inOrderTraversal(root);
    return traversal[traversal.length - k];
}

function inOrderTraversal(root) {
    function helper(node, traversal) {
        if (node) {
            helper(node.left, traversal);
            traversal.push(node.val);
            helper(node.right, traversal);
        }
        return traversal;
    }
    return helper(root, []);
}

function findKthLargestElement(root, k) {
    let stack = [];
    let cur = root;
    let count = 0;
    while (stack || cur) {
        while (cur) {
            stack.push(cur);
            cur = cur.right;
        }
        cur = stack.pop();
        count++;
        if (count === k) {
            return cur.val;
        }
        cur = cur.left;
    }
}

function findKthSmallestElement(root, k) {

function findKthSmallestElementStack(root, k) {
    let stack = [];
    let counter = 0;
    let cur = root;
    while (stack || cur) {
        while (cur) {
            stack.push(cur);
            cur = cur.left;
        }
        cur = stack.pop();
        counter++;
        if (counter == k) {
            return cur.val;
        }
        cur = cur.right;
    }
}
