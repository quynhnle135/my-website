Stack Implementation​

export class Stack {
    constructor() {
        this.items = [];
        this.top = null;
    }
    
    isEmpty() {
        if (this.items.length === 0) return true;
        return false;
    }

    getTop() {
        if (this.isEmpty()) return null;
        return this.top;
    }

    size() {
        return this.items.length;
    }
    
    push(value) {
        this.items.push(value);
        this.top = value;
    }

    pop() {
        // If Stack is not empty
        if (this.items.length !== 0) {
            // If Stack has 1 element
            if (this.items.length === 1) {
                this.top = null;
                return this.items.pop();
            }
            else {
                this.top = this.items[this.items.length - 2];
                return this.items.pop();
            }
        // If Stack is empty
        } else {
            return null;
        }
    }
}

Queue Implementation​

export class Queue {
    constructor() {
        this.items = [];
        this.frontIndex = 0;
        this.backIndex = 0;
    }

    isEmpty() {
        if (this.items.length === 0) return true;
        return false;
    }

    size() {
        return this.items.length;
    }

    enqueue(element) {
        this.items[this.backIndex] = element;
        this.backIndex++;
    }

    dequeue() {
        let item = this.items[this.frontIndex];
        delete this.items[this.frontIndex];
        frontIndex++;
        return item;
    }

    getFront() {
        if (this.isEmpty()) return null;
        return this.items[this.frontIndex];
    }
}

Reverse First K Elements​

// Using a stack
function reverseFirstKElementStack(arr, k) {
    let stack = [];
    for (let i = 0; i < k; i++) {
        stack.push(arr[i]);
    }

    for (let i = 0; i < k; i++) {
        arr[i] = stack.pop();
    }
    return arr;
}
 

// Two pointers
function reverseFirstKElementTwoPointers(arr, k) {
    let start = 0;
    let end = k - 1;
    while (start < end) {
        let temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;
        start++;
        end--;
    }
    return arr;
}

Check Balanced Parentheses​

function checkBalancedParentheses(parentheses) {
    let openings = [];
    for (let p of parentheses) {
        if (p == "(" || p == "{" || p == "[") {
            openings.push(p);
        } else {
            if (openings.lengths === 0) return false;
            else {
                let top = openings.pop();
                if (top === "(" && p !== ")") return false;
                if (top === "[" && p !== "]") return false;
                if (top === "{" && p !== "}") return false;
            }
        }
    }
    if (openings.length === 0) {
        return true;
    } else {
        return false;
    }

}

console.log(checkBalancedParentheses("()"))  // true
console.log(checkBalancedParentheses("()[]{}")) // true
console.log(checkBalancedParentheses("[)"))  // false

Evaluate Postfix Expression​

function evaluatePostfixExpression(expression) {
    let stack = [];
    let result = 0;
    for (let i = 0; i < expression.length; i++) {
        let cur = expression[i];
        if ("+-*/".indexOf(cur) !== -1) {
            let secondNum = stack.pop();
            let firstNum = stack.pop(); 
            if (cur === "+") {
                stack.push(firstNum + secondNum); 
            } else if (cur === "-") {
                stack.push(firstNum - secondNum);
            } else if (cur === "*") {
                stack.push(firstNum * secondNum);
            } else {
                stack.push(Math.floor(firstNum / secondNum));
            }
        } else {
            stack.push(parseInt(cur));
        }
    }
    return stack[0];
}

console.log(evaluatePostfixExpression("921*-8-4+")); // 3

Min Stack​

class MinStack {
    constructor() {
        this.stack = [];
        this.minStack = [];
    }

    pop() {
        if (this.stack.length === 0) {
            console.log("Stack is empty");
            return null;
        }
        let res = this.stack.pop();
        if (res === this.minStack[this.minStack.length - 1]) {
            this.minStack.pop();
        }
        return res;
    }

    push(value) {
        this.stack.push(value);
        if (this.minStack.length === 0 || value < this.minStack[this.minStack.length - 1]) {
            this.minStack.push(value);
        }
    }

    min() {
        return this.minStack[this.minStack.length - 1];
    }
}

Next Greater Elements​

function nextGreaterElement(arr) {
    let stack = [];
    let res = [];
    var top;
    for (let i = arr.length - 1; i >= 0; i--) {
        let cur = arr[i];
        if (stack.length > 0) {
            top = stack[stack.length - 1];
            while (top <= cur) {
                if (stack.length === 0) {
                    break;
                }
                stack.pop();
                top = stack[stack.length - 1];
            }
            // stack.push(top);
        }
        if (stack.length == 0) {
            res[i] = -1;
        } else {
            res[i] = stack[stack.length - 1];
        }
        stack.push(cur);
    }
    return res;
}

console.log(nextGreaterElement([4, 6, 3, 2, 8, 1])) // 6, 8, 8, 8, -1, -1

Find Binary Numbers​

function findBin(num) {
    let queue = [];
    queue.push("1");
    let res = [];
    while (res.length < num) {
        res.push(queue[0]);
        let num1 = res[res.length - 1] + "0";
        let num2 = res[res.length - 1] + "1";
        queue.push(num1);
        queue.push(num2);
        queue = queue.slice(1, queue.length);
    }
    return res;
}

console.log(findBin(5)); // 1 10 11 100 101
console.log(findBin(10)); // 1 10 11 100 101 110 111 1000 1001 1010

function sortValues(stack) {
    let res = [];
    var temp;
    while (stack.length !== 0) {
        if (res.length === 0) {
            res.push(stack.pop());
        } else {
            temp = stack.pop();
            if (temp < res[res.length - 1]) {
                res.push(temp);
            } else {
                while (res.length !== 0 && temp >= res[res.length - 1]) {
                    stack.push(res.pop());
                }
                res.push(temp);
            }
        }
    }
    return res;
}
